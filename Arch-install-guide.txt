# Create the partitions in gparted live usb.
For better overview.
(In gparted dont forget to change to gpt
and also change bootflag to boot to get efi when finish)

# Keyboard
loadkeys sv-latin1

# Testing efi
ls /sys/firmware/efi/efivars

# Network 
ip link

ping archlinux.org

timedatectl

# Mount partitions
mount /dev/sda3 /mnt
mkdir /mnt/home
mkdir -p /mnt/boot/efi
mount /dev/sda4 /mnt/home
mount /dev/sda1 /mnt/boot/efi
swapon /dev/sda2

# edit mirrorlist
nano /etc/pacman.d/mirrorlist

# install base
pacstrap -K /mnt base linux linux-firmware intel-ucode 

# create fstab
genfstab -U /mnt >> /mnt/etc/fstab

# Into the abyss
arch-chroot /mnt

# Add some useful tools
pacman -S base-devel nano git grub efibootmgr networkmanager-qt sudo 

# timezone
ln -sf /usr/share/zoneinfo/Europe/Stockholm /etc/localtime

# hwclock
hwclock --systohc

# language
locale-gen

# nano /etc/locale.conf
LANG=en_US.UTF-8

# nano /etc/vconsole.conf
KEYMAP=sv-latin1

echo MYHOSTNAME > /etc/hostname
# nano /etc/hostname
box

# kernel config 
mkinitcpio -P

# set password
passwd

# Enable services
systemctl enable sddm.service

systemctl enable NetworkManager.service

# Add user
useradd -m -G wheel -s /bin/bash username
passwd username

EDITOR=nano visudo
# uncomment %wheel ALL=(ALL) ALL
# without password
username  ALL=(ALL) NOPASSWD:ALL

grub-install --target=x86_64-efi --efi-directory=/boot/efi --bootloader-id=GRUB
grub-mkconfig -o /boot/grub/grub.cfg

# nano etc/mkinitcpio.conf
Mkinitcpio:
HOOKS=(base systemd autodetect keyboard sd-vconsole modconf block ext4 filesystems fsck)
MODULES="ahci usbhid xhci_hcd sd_mod ehci-hcd ohci-hcd ext4 vfat"
BINARIES="fsck fsck.ext4"

pacman -S --needed git base-devel
git clone https://aur.archlinux.org/yay-bin.git
cd yay-bin
makepkg -si

yay -S microsoft-edge-stable-bin

pacman -S  plasma-desktop plasma-wayland-session pipewire sddm dolphin bluez-utils blueman konsole plasma-nm mesa chromium
ttf-jetbrains-mono-nerd noto-fonts-emoji

# Breezetheme
sudo nano /usr/lib/sddm/sddm.conf.d/default.conf
